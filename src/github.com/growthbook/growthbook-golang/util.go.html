    29  // This converts an experiment's coverage and variation weights into
    30  // an array of bucket ranges.
    31  func getBucketRanges(numVariations int, coverage float64, weights []float64) []VariationRange {
    32  	// Make sure coverage is within bounds.
    33  	if coverage < 0 {
    34  		logWarn(WarnExpCoverageMustBePositive)
    35  		coverage = 0
    36  	}
    37  	if coverage > 1 {
    38  		logWarn(WarnExpCoverageMustBeFraction)
    39  		coverage = 1
    40  	}
    41  
    42  	// Default to equal weights if missing or invalid
    43  	if weights == nil || len(weights) == 0 {
    44  		weights = getEqualWeights(numVariations)
    45  	}
    46  	if len(weights) != numVariations {
    47  		logWarn(WarnExpWeightsWrongLength)
    48  		weights = getEqualWeights(numVariations)
    49  	}
    50  
    51  	// If weights don't add up to 1 (or close to it), default to equal weights
    52  	totalWeight := 0.0
    53  	for i := range weights {
    54  		totalWeight += weights[i]
    55  	}
    56  	if totalWeight < 0.99 || totalWeight > 1.01 {
    57  		logWarn(WarnExpWeightsWrongTotal)
    58  		weights = getEqualWeights(numVariations)
    59  	}
    60  
    61  	// Convert weights to ranges
    62  	cumulative := 0.0
    63  	ranges := make([]VariationRange, len(weights))
    64  	for i := range weights {
    65  		start := cumulative
    66  		cumulative += weights[i]
    67  		ranges[i] = VariationRange{start, start + coverage*weights[i]}
    68  	}
    69  	return ranges
    70  }

   120  // Simple wrapper around Go standard library FNV32a hash function.
   121  func hashFnv32a(s string) uint32 {
   122  	hash := fnv.New32a()
   123  	hash.Write([]byte(s))
   124  	return hash.Sum32()

   129  func truthy(v interface{}) bool {

   131  		return false
   132  	}
   133  	switch v.(type) {

   135  		return v.(string) != ""
   136  	case bool:
   137  		return v.(bool)

   139  		return v.(int) != 0
   140  	case uint:
   141  		return v.(uint) != 0
   142  	case float32:

   144  	case float64:

   146  	}

   150  // This function converts slices of concrete types to []interface{}.
   151  // This is needed to handle the common case where a user passes an
   152  // attribute as a []string (or []int), and this needs to be compared
   153  // against feature data deserialized from JSON, which always results
   154  // in []interface{} slices.
   155  func fixSliceTypes(vin interface{}) interface{} {
   156  	// Convert all type-specific slices to interface{} slices.
   157  	v := reflect.ValueOf(vin)
   158  	rv := vin
   159  	if v.Kind() == reflect.Slice {
   160  		srv := make([]interface{}, v.Len())
   161  		for i := 0; i < v.Len(); i++ {
   162  			elem := v.Index(i).Interface()
   163  			srv[i] = elem
   164  		}
   165  		rv = srv
   166  	}
   167  	return rv
   168  }
