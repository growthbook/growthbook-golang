    30  // This converts an experiment's coverage and variation weights into
    31  // an array of bucket ranges.
    32  func getBucketRanges(numVariations int, coverage float64, weights []float64) []VariationRange {
    33  	// Make sure coverage is within bounds.
    34  	if coverage < 0 {
    35  		logWarn(WarnExpCoverageMustBePositive)
    36  		coverage = 0
    37  	}
    38  	if coverage > 1 {
    39  		logWarn(WarnExpCoverageMustBeFraction)
    40  		coverage = 1
    41  	}
    42  
    43  	// Default to equal weights if missing or invalid
    44  	if weights == nil || len(weights) == 0 {
    45  		weights = getEqualWeights(numVariations)
    46  	}
    47  	if len(weights) != numVariations {
    48  		logWarn(WarnExpWeightsWrongLength)
    49  		weights = getEqualWeights(numVariations)
    50  	}
    51  
    52  	// If weights don't add up to 1 (or close to it), default to equal weights
    53  	totalWeight := 0.0
    54  	for i := range weights {
    55  		totalWeight += weights[i]
    56  	}
    57  	if totalWeight < 0.99 || totalWeight > 1.01 {
    58  		logWarn(WarnExpWeightsWrongTotal)
    59  		weights = getEqualWeights(numVariations)
    60  	}
    61  
    62  	// Convert weights to ranges
    63  	cumulative := 0.0
    64  	ranges := make([]VariationRange, len(weights))
    65  	for i := range weights {
    66  		start := cumulative
    67  		cumulative += weights[i]
    68  		ranges[i] = VariationRange{start, start + coverage*weights[i]}
    69  	}
    70  	return ranges
    71  }

   121  // Convert integer or string hash values to strings.
   122  func convertHashValue(vin interface{}) (string, bool) {
   123  	hashString, stringOK := vin.(string)
   124  	if stringOK {
   125  		if hashString == "" {
   126  			logInfo(InfoRuleSkipEmptyHashAttribute)
   127  			return "", false
   128  		}
   129  		return hashString, true
   130  	}
   131  	hashInt, intOK := vin.(int)
   132  	if intOK {
   133  		return fmt.Sprint(hashInt), true
   134  	}
   135  	hashFloat, floatOK := vin.(float64)
   136  	if floatOK {
   137  		return fmt.Sprint(int(hashFloat)), true
   138  	}
   139  	return "", false

   172  // This function converts slices of concrete types to []interface{}.
   173  // This is needed to handle the common case where a user passes an
   174  // attribute as a []string (or []int), and this needs to be compared
   175  // against feature data deserialized from JSON, which always results
   176  // in []interface{} slices.
   177  func fixSliceTypes(vin interface{}) interface{} {
   178  	// Convert all type-specific slices to interface{} slices.
   179  	v := reflect.ValueOf(vin)
   180  	rv := vin
   181  	if v.Kind() == reflect.Slice || v.Kind() == reflect.Array {
   182  		srv := make([]interface{}, v.Len())
   183  		for i := 0; i < v.Len(); i++ {
   184  			elem := v.Index(i).Interface()
   185  			srv[i] = elem
   186  		}
   187  		rv = srv
   188  	}
   189  	return rv
   190  }
