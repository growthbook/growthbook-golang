    28  // This converts an experiment's coverage and variation weights into
    29  // an array of bucket ranges.
    30  func getBucketRanges(numVariations int, coverage float64, weights []float64) []VariationRange {
    31  	// Make sure coverage is within bounds.
    32  	if coverage < 0 {
    33  		logWarn(WarnExpCoverageMustBePositive)
    34  		coverage = 0
    35  	}
    36  	if coverage > 1 {
    37  		logWarn(WarnExpCoverageMustBeFraction)
    38  		coverage = 1
    39  	}
    40  
    41  	// Default to equal weights if missing or invalid
    42  	if weights == nil || len(weights) == 0 {
    43  		weights = getEqualWeights(numVariations)
    44  	}
    45  	if len(weights) != numVariations {
    46  		logWarn(WarnExpWeightsWrongLength)
    47  		weights = getEqualWeights(numVariations)
    48  	}
    49  
    50  	// If weights don't add up to 1 (or close to it), default to equal weights
    51  	totalWeight := 0.0
    52  	for i := range weights {
    53  		totalWeight += weights[i]
    54  	}
    55  	if totalWeight < 0.99 || totalWeight > 1.01 {
    56  		logWarn(WarnExpWeightsWrongTotal)
    57  		weights = getEqualWeights(numVariations)
    58  	}
    59  
    60  	// Convert weights to ranges
    61  	cumulative := 0.0
    62  	ranges := make([]VariationRange, len(weights))
    63  	for i := range weights {
    64  		start := cumulative
    65  		cumulative += weights[i]
    66  		ranges[i] = VariationRange{start, start + coverage*weights[i]}
    67  	}
    68  	return ranges
    69  }
