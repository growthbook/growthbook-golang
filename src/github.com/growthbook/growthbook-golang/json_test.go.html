   260  // Run a set of JSON test cases.
   261  func jsonTest(t *testing.T, label string,
   262  	fn func(g *G, itest int, test []interface{})) {
   263  	content, err := ioutil.ReadFile("cases.json")
   264  	if err != nil {
   265  		log.Fatal(err)
   266  	}
   267  
   268  	// Unmarshal all test cases at once.
   269  	allCases := map[string]interface{}{}
   270  	err = json.Unmarshal(content, &allCases)
   271  	if err != nil {
   272  		log.Fatal(err)
   273  	}
   274  
   275  	// Extract just the test cases for the test type we're working on.
   276  	cases := allCases[label].([]interface{})
   277  
   278  	// Extract the test data for each case as a JSON array and pass to
   279  	// the test function.
   280  	g := Goblin(t)
   281  	g.Describe("json test suite: "+label, func() {
   282  		// Handle logging during tests: reset log before each test, make
   283  		// sure there are no errors, and make sure there is never more
   284  		// than one warning during a test (some tests that check for
   285  		// correct handling of out-of-range parameters trigger warnings,
   286  		// but there should never be more than one per test).
   287  		g.BeforeEach(func() { testLog.reset() })
   288  		g.AfterEach(func() {
   289  			g.Assert(len(testLog.errors)).Equal(0, "test log has errors:", testLog.allErrors())
   290  			g.Assert(len(testLog.warnings) <= 1).IsTrue("test log has warnings:", testLog.allWarnings())
   291  		})
   292  
   293  		// Run tests one at a time: each test's JSON data is an array,
   294  		// with the interpretation of the array entries depending on the
   295  		// test type.
   296  		for itest, gtest := range cases {
   297  			test, ok := gtest.([]interface{})
   298  			if !ok {
   299  				log.Fatal("unpacking JSON test data")
   300  			}
   301  			fn(g, itest, test)
   302  		}
   303  	})
   304  }

   306  // Helper to round variation ranges for comparison with fixed test
   307  // values.
   308  func roundRanges(ranges []VariationRange) []VariationRange {
   309  	result := make([]VariationRange, len(ranges))
   310  	for i, r := range ranges {
   311  		rmin := math.Round(r.Min*1000000) / 1000000
   312  		rmax := math.Round(r.Max*1000000) / 1000000
   313  		result[i] = VariationRange{rmin, rmax}
   314  	}
   315  	return result
   316  }
