   258  // Run a set of JSON test cases.
   259  func jsonTest(t *testing.T, label string,
   260  	fn func(g *G, itest int, test []interface{})) {
   261  	content, err := ioutil.ReadFile("cases.json")
   262  	if err != nil {
   263  		log.Fatal(err)
   264  	}
   265  
   266  	// Unmarshal all test cases at once.
   267  	allCases := map[string]interface{}{}
   268  	err = json.Unmarshal(content, &allCases)
   269  	if err != nil {
   270  		log.Fatal(err)
   271  	}
   272  
   273  	// Extract just the test cases for the test type we're working on.
   274  	cases := allCases[label].([]interface{})
   275  
   276  	// Extract the test data for each case as a JSON array and pass to
   277  	// the test function.
   278  	g := Goblin(t)
   279  	g.Describe("json test suite: "+label, func() {
   280  		// Handle logging during tests: reset log before each test, make
   281  		// sure there are no errors, and make sure there is never more
   282  		// than one warning during a test (some tests that check for
   283  		// correct handling of out-of-range parameters trigger warnings,
   284  		// but there should never be more than one per test).
   285  		g.BeforeEach(func() { testLog.reset() })
   286  		g.AfterEach(func() {
   287  			g.Assert(len(testLog.errors)).Equal(0)
   288  			g.Assert(len(testLog.warnings) <= 1).IsTrue()
   289  		})
   290  
   291  		// Run tests one at a time: each test's JSON data is an array,
   292  		// with the interpretation of the array entries depending on the
   293  		// test type.
   294  		for itest, gtest := range cases {
   295  			test, ok := gtest.([]interface{})
   296  			if !ok {
   297  				log.Fatal("unpacking JSON test data")
   298  			}
   299  			fn(g, itest, test)
   300  		}
   301  	})
   302  }

   304  // Helper to round variation ranges for comparison with fixed test
   305  // values.
   306  func roundRanges(ranges []VariationRange) []VariationRange {
   307  	result := make([]VariationRange, len(ranges))
   308  	for i, r := range ranges {
   309  		rmin := math.Round(r.Min*1000000) / 1000000
   310  		rmax := math.Round(r.Max*1000000) / 1000000
   311  		result[i] = VariationRange{rmin, rmax}
   312  	}
   313  	return result
   314  }
