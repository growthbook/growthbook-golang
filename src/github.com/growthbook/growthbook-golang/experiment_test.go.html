     9  func TestSubscriptions(t *testing.T) {
    10  	g := Goblin(t)
    11  
    12  	g.Describe("subscriptions", func() {
    13  		var context *Context
    14  		var gb *GrowthBook
    15  		var exp1 *Experiment
    16  		var exp2 *Experiment
    17  
    18  		g.BeforeEach(func() {
    19  			context = NewContext().WithAttributes(Attributes{"id": "1"})
    20  			gb = New(context)
    21  			exp1 = NewExperiment("experiment-1").WithVariations("result1", "result2")
    22  			exp2 = NewExperiment("experiment-2").WithVariations("result3", "result4")
    23  		})
    24  
    25  		g.It("can subscribe to experiments", func() {
    26  			var savedExp *Experiment
    27  			called := 0
    28  			gb.Subscribe(func(exp *Experiment, result *ExperimentResult) {
    29  				savedExp = exp
    30  				called++
    31  			})
    32  
    33  			gb.Run(exp1)
    34  			gb.Run(exp1)
    35  
    36  			g.Assert(savedExp).Equal(exp1)
    37  			// Subscription only gets triggered once for repeated experiment
    38  			// runs.
    39  			g.Assert(called).Equal(1)
    40  
    41  			savedExp = nil
    42  			called = 0
    43  
    44  			gb.ClearSavedResults()
    45  			gb.Run(exp1)
    46  			// Change attributes to change experiment result so subscription
    47  			// gets triggered twice.
    48  			gb.WithAttributes(Attributes{"id": "3"})
    49  			gb.Run(exp1)
    50  
    51  			g.Assert(savedExp).Equal(exp1)
    52  			g.Assert(called).Equal(2)
    53  		})
    54  
    55  		g.It("can unsubscribe from experiments", func() {
    56  			var savedExp *Experiment
    57  			called := 0
    58  			unsubscribe := gb.Subscribe(func(exp *Experiment, result *ExperimentResult) {
    59  				savedExp = exp
    60  				called++
    61  			})
    62  
    63  			gb.Run(exp1)
    64  			gb.WithAttributes(Attributes{"id": "3"})
    65  			unsubscribe()
    66  			gb.Run(exp1)
    67  
    68  			g.Assert(savedExp).Equal(exp1)
    69  			g.Assert(called).Equal(1)
    70  		})
    71  
    72  		g.It("can track experiment results", func() {
    73  			called := 0
    74  			context.WithTrackingCallback(func(exp *Experiment, result *ExperimentResult) {
    75  				called++
    76  			})
    77  
    78  			gb.Run(exp1)
    79  			gb.Run(exp2)
    80  			gb.Run(exp1)
    81  			gb.Run(exp2)
    82  			gb.WithAttributes(Attributes{"id": "3"})
    83  			gb.Run(exp1)
    84  			gb.Run(exp2)
    85  			gb.Run(exp1)
    86  			gb.Run(exp2)
    87  			g.Assert(called).Equal(4)
    88  		})
    89

    92  			gb.Run(exp2)
    93  			g.Assert(len(gb.GetAllResults())).Equal(2)
    94  		})
