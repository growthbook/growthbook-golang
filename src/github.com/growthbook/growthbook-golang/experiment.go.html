    86  // BuildExperiment creates an Experiment value from a JSON object
    87  // represented as a Go map.
    88  func BuildExperiment(dict map[string]interface{}) *Experiment {
    89  	exp := NewExperiment("tmp")
    90  	gotKey := false
    91  	for k, v := range dict {
    92  		switch k {
    93  		case "key":
    94  			tmp, ok := v.(string)
    95  			if !ok {
    96  				logError(ErrJSONInvalidType, "Experiment", "key")
    97  				continue
    98  			}
    99  			exp.Key = tmp
   100  			gotKey = true
   101  		case "variations":
   102  			exp = exp.WithVariations(BuildFeatureValues(v)...)
   103  		case "weights":
   104  			vals, ok := v.([]interface{})
   105  			if !ok {
   106  				logError(ErrJSONInvalidType, "Experiment", "weights")
   107  				continue
   108  			}
   109  			weights := make([]float64, len(vals))
   110  			for i := range vals {
   111  				val, ok := vals[i].(float64)
   112  				if !ok {
   113  					logError(ErrJSONInvalidType, "Experiment", "weights")
   114  					continue
   115  				}
   116  				weights[i] = val
   117  			}
   118  			exp = exp.WithWeights(weights...)
   119  		case "active":
   120  			tmp, ok := v.(bool)
   121  			if !ok {
   122  				logError(ErrJSONInvalidType, "Experiment", "key")
   123  				continue
   124  			}
   125  			exp = exp.WithActive(tmp)
   126  		case "coverage":
   127  			tmp, ok := v.(float64)
   128  			if !ok {
   129  				logError(ErrJSONInvalidType, "Experiment", "coverage")
   130  				continue
   131  			}
   132  			exp = exp.WithCoverage(tmp)
   133  		case "condition":
   134  			tmp, ok := v.(map[string]interface{})
   135  			if !ok {
   136  				logError(ErrJSONInvalidType, "Experiment", "condition")
   137  				continue
   138  			}
   139  			cond := BuildCondition(tmp)
   140  			if cond == nil {
   141  				logError(ErrExpJSONInvalidCondition)
   142  			} else {
   143  				exp = exp.WithCondition(cond)
   144  			}
   145  		case "namespace":
   146  			exp = exp.WithNamespace(BuildNamespace(v))
   147  		case "force":
   148  			tmp, ok := v.(float64)
   149  			if !ok {
   150  				logError(ErrJSONInvalidType, "Experiment", "force")
   151  				continue
   152  			}
   153  			exp = exp.WithForce(int(tmp))
   154  		case "hashAttribute":
   155  			tmp, ok := v.(string)
   156  			if !ok {
   157  				logError(ErrJSONInvalidType, "Experiment", "hashAttribute")
   158  				continue
   159  			}
   160  			exp = exp.WithHashAttribute(tmp)
   161  		default:
   162  			logWarn(WarnJSONUnknownKey, "Experiment", k)
   163  		}
   164  	}
   165  	if !gotKey {
   166  		logWarn(WarnExpJSONKeyNotSet)
   167  	}
   168  	return exp
   169  }
