   100  // Feature returns the result for a feature identified by a string
   101  // feature key.
   102  func (gb *GrowthBook) Feature(key string) *FeatureResult {
   103  	// Handle unknown features.
   104  	feature, ok := gb.context.Features[key]
   105  	if !ok {
   106  		return getFeatureResult(nil, UnknownFeatureResultSource, nil, nil)
   107  	}
   108  
   109  	// Loop through the feature rules (if any).
   110  	for _, rule := range feature.Rules {
   111  
   112  		// If the rule has a condition and the condition does not pass,
   113  		// skip this rule.
   114  		if rule.Condition != nil && !rule.Condition.Eval(gb.Attributes()) {
   115  			logInfo(InfoRuleSkipCondition, key, rule)
   116  			continue
   117  		}
   118  
   119  		// If rule.Force has been set:
   120  		if rule.Force != nil {
   121  			// If rule.Coverage is set
   122  			if rule.Coverage != nil {
   123  				// Get the value of the hashAttribute, defaulting to "id", and
   124  				// if missing or empty, skip the rule.
   125  				hashAttribute := "id"
   126  				if rule.HashAttribute != nil {
   127  					hashAttribute = *rule.HashAttribute
   128  				}
   129  				hashValue, ok := gb.Attributes()[hashAttribute]
   130  				if !ok {
   131  					logInfo(InfoRuleSkipNoHashAttribute, key, rule)
   132  					continue
   133  				}
   134  				hashString, ok := convertHashValue(hashValue)
   135  				if !ok {
   136  					continue
   137  				}
   138  
   139  				// Hash the value.
   140  				n := float64(hashFnv32a(hashString+key)%1000) / 1000
   141  
   142  				// If the hash is greater than rule.Coverage, skip the rule.
   143  				if n > *rule.Coverage {
   144  					logInfo(InfoRuleSkipCoverage, key, rule)
   145  					continue
   146  				}
   147  			}
   148  
   149  			// Return forced feature result.
   150  			return getFeatureResult(rule.Force, ForceResultSource, nil, nil)
   151  		}
   152  
   153  		// Otherwise, convert the rule to an Experiment object, copying
   154  		// values from the rule as necessary.
   155  		experiment := Experiment{
   156  			Key:        key,
   157  			Variations: rule.Variations,
   158  			Active:     true,
   159  		}
   160  		if rule.TrackingKey != nil {
   161  			experiment.Key = *rule.TrackingKey
   162  		}
   163  		if rule.Coverage != nil {
   164  			var tmp *float64
   165  			if rule.Coverage != nil {
   166  				val := *rule.Coverage
   167  				tmp = &val
   168  			}
   169  			experiment.Coverage = tmp
   170  		}
   171  		if rule.Weights != nil {
   172  			var tmp []float64
   173  			if rule.Weights != nil {
   174  				tmp = make([]float64, len(rule.Weights))
   175  				copy(tmp, rule.Weights)
   176  			}
   177  			experiment.Weights = tmp
   178  		}
   179  		if rule.HashAttribute != nil {
   180  			var tmp *string
   181  			if rule.HashAttribute != nil {
   182  				val := *rule.HashAttribute
   183  				tmp = &val
   184  			}
   185  			experiment.HashAttribute = tmp
   186  		}
   187  		if rule.Namespace != nil {
   188  			var tmp *Namespace
   189  			if rule.Namespace != nil {
   190  				val := Namespace{rule.Namespace.ID, rule.Namespace.Start, rule.Namespace.End}
   191  				tmp = &val
   192  			}
   193  			experiment.Namespace = tmp
   194  		}
   195  
   196  		// Run the experiment.
   197  		result := gb.doRun(&experiment, &key)
   198  
   199  		// If result.inExperiment is false, skip this rule and continue.
   200  		if !result.InExperiment {
   201  			logInfo(InfoRuleSkipUserNotInExp, key, rule)
   202  			continue
   203  		}
   204  
   205  		// Otherwise, return experiment result.
   206  		return getFeatureResult(result.Value, ExperimentResultSource, &experiment, result)
   207  	}
   208  
   209  	// Fall back to using the default value
   210  	return getFeatureResult(feature.DefaultValue, DefaultValueResultSource, nil, nil)
   211  }

   227  func (gb *GrowthBook) getExperimentResult(
   228  	exp *Experiment, variationIndex int,
   229  	hashUsed bool, featureID *string) *ExperimentResult {
   230  	inExperiment := true
   231  
   232  	// If assigned variation is not valid, use the baseline and mark the
   233  	// user as not in the experiment
   234  	if variationIndex < 0 || variationIndex >= len(exp.Variations) {
   235  		variationIndex = 0
   236  		inExperiment = false
   237  	}
   238  
   239  	// Get the hashAttribute and hashValue
   240  	hashAttribute := "id"
   241  	if exp.HashAttribute != nil {
   242  		hashAttribute = *exp.HashAttribute
   243  	}
   244  	hashString := ""
   245  	hashValue, ok := gb.context.Attributes[hashAttribute]
   246  	if ok {
   247  		hashString, _ = convertHashValue(hashValue)
   248  	}
   249  
   250  	// Return
   251  	var value FeatureValue
   252  	if variationIndex < len(exp.Variations) {
   253  		value = exp.Variations[variationIndex]
   254  	}
   255  	return &ExperimentResult{
   256  		InExperiment:  inExperiment,
   257  		FeatureID:     featureID,
   258  		VariationID:   variationIndex,
   259  		Value:         value,
   260  		HashUsed:      hashUsed,
   261  		HashAttribute: hashAttribute,
   262  		HashValue:     hashString,
   263  	}
   264  }

   266  // Run an experiment. (Uses doRun to make wrapping for subscriptions
   267  // simple.)
   268  func (gb *GrowthBook) Run(exp *Experiment) *ExperimentResult {
   269  	// Actually run the experiment.
   270  	result := gb.doRun(exp, nil)
   271  
   272  	// Determine whether the result changed from the last stored result
   273  	// for the experiment.
   274  	changed := false
   275  	storedResult, exists := gb.latestResults[exp.Key]
   276  	if exists {
   277  		if storedResult.InExperiment != result.InExperiment ||
   278  			storedResult.VariationID != result.VariationID {
   279  			changed = true
   280  		}
   281  	}
   282  
   283  	// Store the experiment result.
   284  	gb.latestResults[exp.Key] = result
   285  
   286  	// If the result changed, trigger all subscriptions.
   287  	if changed || !exists {
   288  		for _, sub := range gb.subscriptions {
   289  			sub(exp, result)
   290  		}
   291  	}
   292  
   293  	return result
   294  }

   296  // Worker function to run an experiment.
   297  func (gb *GrowthBook) doRun(exp *Experiment, featureID *string) *ExperimentResult {
   298  	// 1. If exp.Variations has fewer than 2 variations, return default
   299  	//    result.
   300  	if len(exp.Variations) < 2 {
   301  		return gb.getExperimentResult(exp, -1, false, featureID)
   302  	}
   303  
   304  	// 2. If context.Enabled is false, return default result.
   305  	if !gb.context.Enabled {
   306  		return gb.getExperimentResult(exp, -1, false, featureID)
   307  	}
   308  
   309  	// 3. If context.URL exists, check for query string override and use
   310  	//    it if it exists.
   311  	if gb.context.URL != nil {
   312  		qsOverride := getQueryStringOverride(exp.Key, gb.context.URL, len(exp.Variations))
   313  		if qsOverride != nil {
   314  			return gb.getExperimentResult(exp, *qsOverride, false, featureID)
   315  		}
   316  	}
   317  
   318  	// 4. Return forced result if forced via context.
   319  	force, forced := gb.context.ForcedVariations[exp.Key]
   320  	if forced {
   321  		return gb.getExperimentResult(exp, force, false, featureID)
   322  	}
   323  
   324  	// 5. If exp.Active is set to false, return default result.
   325  	if !exp.Active {
   326  		return gb.getExperimentResult(exp, -1, false, featureID)
   327  	}
   328  
   329  	// 6. Get the user hash value and return if empty.
   330  	hashAttribute := "id"
   331  	if exp.HashAttribute != nil {
   332  		hashAttribute = *exp.HashAttribute
   333  	}
   334  	hashString := ""
   335  	hashValue, ok := gb.context.Attributes[hashAttribute]
   336  	if ok {
   337  		hashString, _ = convertHashValue(hashValue)
   338  	}
   339  	if hashString == "" {
   340  		return gb.getExperimentResult(exp, -1, false, featureID)
   341  	}
   342  
   343  	// 7. If exp.Namespace is set, return if not in range.
   344  	if exp.Namespace != nil {
   345  		if !inNamespace(hashString, exp.Namespace) {
   346  			return gb.getExperimentResult(exp, -1, false, featureID)
   347  		}
   348  	}
   349  
   350  	// 8. If exp.Condition is set, return if it evaluates to false.
   351  	if exp.Condition != nil {
   352  		if !exp.Condition.Eval(gb.context.Attributes) {
   353  			return gb.getExperimentResult(exp, -1, false, featureID)
   354  		}
   355  	}
   356  
   357  	// 9. Calculate bucket ranges for the variations and choose one.
   358  	coverage := float64(1)
   359  	if exp.Coverage != nil {
   360  		coverage = *exp.Coverage
   361  	}
   362  	ranges := getBucketRanges(len(exp.Variations), coverage, exp.Weights)
   363  	n := float64(hashFnv32a(hashString+exp.Key)%1000) / 1000
   364  	assigned := chooseVariation(float64(n), ranges)
   365  
   366  	// 10. If assigned == -1, return default result.
   367  	if assigned == -1 {
   368  		return gb.getExperimentResult(exp, -1, false, featureID)
   369  	}
   370  
   371  	// 11. If experiment has a forced variation, return it.
   372  	if exp.Force != nil {
   373  		return gb.getExperimentResult(exp, *exp.Force, false, featureID)
   374  	}
   375  
   376  	// 12. If in QA mode, return default result.
   377  	if gb.context.QAMode {
   378  		return gb.getExperimentResult(exp, -1, false, featureID)
   379  	}
   380  
   381  	// 13. Build the result object.
   382  	result := gb.getExperimentResult(exp, assigned, true, featureID)
   383  
   384  	// 14. Fire tracking callback if required.
   385  	gb.track(exp, result)
   386  
   387  	return result
   388  }

   390  // Fire Context.TrackingCallback if it's set and the combination of
   391  // hashAttribute, hashValue, experiment key, and variation ID has not
   392  // been tracked before.
   393  func (gb *GrowthBook) track(exp *Experiment, result *ExperimentResult) {
   394  	if gb.context.TrackingCallback == nil {
   395  		return
   396  	}
   397  
   398  	// Make sure tracking callback is only fired once per unique
   399  	// experiment.
   400  	key := result.HashAttribute + result.HashValue +
   401  		exp.Key + strconv.Itoa(result.VariationID)
   402  	if _, exists := gb.trackedExperiments[key]; exists {
   403  		return
   404  	}
   405  
   406  	gb.trackedExperiments[key] = true
   407  	gb.context.TrackingCallback(exp, result)
   408  }

   410  // Subscribe adds a callback that is called every time GrowthBook.Run
   411  // is called. This is different from the tracking callback since it
   412  // also fires when a user is not included in an experiment.
   413  func (gb *GrowthBook) Subscribe(callback ExperimentCallback) func() {
   414  	id := gb.nextSubscriptionID
   415  	gb.subscriptions[id] = callback
   416  	gb.nextSubscriptionID++
   417  	return func() {
   418  		delete(gb.subscriptions, id)
   419  	}
   420  }
