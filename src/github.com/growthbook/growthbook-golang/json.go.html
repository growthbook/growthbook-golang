   229  func BuildNamespace(val interface{}) *Namespace {
   230  	array, ok := val.([]interface{})
   231  	if !ok || len(array) != 3 {
   232  		return nil
   233  	}
   234  	id, ok1 := array[0].(string)
   235  	start, ok2 := array[1].(float64)
   236  	end, ok3 := array[2].(float64)
   237  	if !ok1 || !ok2 || !ok3 {
   238  		return nil
   239  	}
   240  	return &Namespace{id, start, end}
