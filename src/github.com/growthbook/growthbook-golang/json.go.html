   215  func BuildNamespace(val interface{}) *Namespace {
   216  	array, ok := val.([]interface{})
   217  	if !ok || len(array) != 3 {
   218  		return nil
   219  	}
   220  	id, ok1 := array[0].(string)
   221  	start, ok2 := array[1].(float64)
   222  	end, ok3 := array[2].(float64)
