    42  // WithAttributes sets the attributes for a context.
    43  func (ctx *Context) WithAttributes(attributes Attributes) *Context {
    44  	savedAttributes := Attributes{}
    45  	for k, v := range attributes {
    46  		// Skip attributes that are arrays: not allowed.
    47  		if reflect.ValueOf(v).Kind() == reflect.Array {
    48  			ctx.Valid = false
    49  			logError(ErrCtxArrayInAttributes, k)
    50  			continue
    51  		}
    52  		savedAttributes[k] = fixSliceTypes(v)
    53  	}
    54  	ctx.Attributes = savedAttributes
    55  	return ctx
    56  }

    94  func ParseContext(data []byte) *Context {
    95  	dict := map[string]interface{}{}
    96  	err := json.Unmarshal(data, &dict)
    97  	if err != nil {
    98  		logError(ErrJSONFailedToParse, "Context")
    99  		return NewContext()

   104  // BuildContext creates a Context value from a JSON object represented
   105  // as a Go map.
   106  func BuildContext(dict map[string]interface{}) *Context {
   107  	context := NewContext()
   108  	for k, v := range dict {
   109  		switch k {
   110  		case "enabled":
   111  			context = context.WithEnabled(v.(bool))
   112  		case "attributes":
   113  			context = context.WithAttributes(v.(map[string]interface{}))
   114  		case "url":
   115  			url, err := url.Parse(v.(string))
   116  			if err != nil {
   117  				logError(ErrCtxJSONInvalidURL, v.(string))
   118  			} else {
   119  				context = context.WithURL(url)
   120  			}
   121  		case "features":
   122  			context.Features = BuildFeatureMap(v.(map[string]interface{}))
   123  		case "forcedVariations":
   124  			vars := map[string]int{}
   125  			for k, vr := range v.(map[string]interface{}) {
   126  				vars[k] = int(vr.(float64))
   127  			}
   128  			context = context.WithForcedVariations(vars)
   129  		case "qaMode":
   130  			context = context.WithQAMode(v.(bool))
   131  		default:
   132  			logWarn(WarnJSONUnknownKey, "Context", k)
   133  		}
   134  	}
   135  	return context
   136  }
