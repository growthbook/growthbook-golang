    20  // ExperimentCallback is a callback function that is executed every
    21  // time a user is included in an Experiment. It is also the type used
    22  // for subscription functions, which are called whenever
    23  // Experiment.Run is called and the experiment result changes,
    24  // independent of whether a user is inncluded in the experiment or

    81  // ParseContext creates a Context value from raw JSON input.
    82  func ParseContext(data []byte) *Context {
    83  	dict := map[string]interface{}{}
    84  	err := json.Unmarshal(data, &dict)
    85  	if err != nil {
    86  		logError(ErrJSONFailedToParse, "Context")
    87  		return NewContext()
    88  	}
    89  	return BuildContext(dict)
    90  }

    92  // BuildContext creates a Context value from a JSON object represented
    93  // as a Go map.
    94  func BuildContext(dict map[string]interface{}) *Context {
    95  	context := NewContext()
    96  	for k, v := range dict {
    97  		switch k {
    98  		case "enabled":
    99  			context = context.WithEnabled(v.(bool))
   100  		case "attributes":
   101  			context = context.WithAttributes(v.(map[string]interface{}))
   102  		case "url":
   103  			url, err := url.Parse(v.(string))
   104  			if err != nil {
   105  				logError(ErrCtxJSONInvalidURL, v.(string))
   106  			} else {
   107  				context = context.WithURL(url)
   108  			}
   109  		case "features":
   110  			context.Features = BuildFeatureMap(v.(map[string]interface{}))
   111  		case "forcedVariations":
   112  			vars := map[string]int{}
   113  			for k, vr := range v.(map[string]interface{}) {
   114  				vars[k] = int(vr.(float64))
   115  			}
   116  			context = context.WithForcedVariations(vars)
   117  		case "qaMode":
   118  			context = context.WithQAMode(v.(bool))
   119  		default:
   120  			logWarn(WarnJSONUnknownKey, "Context", k)
   121  		}
   122  	}
   123  	return context
   124  }
