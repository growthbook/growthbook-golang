    51  func setupWithDelay(provideSSE bool, delay time.Duration, encryptedFeatures string) *env {
    52  	SetLogger(&testLog)
    53  	testLog.reset()
    54  
    55  	initialValue := "initial"
    56  	callCount := 0
    57  	urls := map[string]int{}
    58  	env := env{featureValue: &initialValue, callCount: &callCount, urls: urls}
    59  
    60  	// We need to set up a mock server to handle normal API requests and
    61  	// SSE updates.
    62  	mux := http.NewServeMux()
    63  
    64  	// Normal GET.
    65  	mux.HandleFunc("/api/features/", func(w http.ResponseWriter, r *http.Request) {
    66  		env.Lock()
    67  		defer env.Unlock()
    68  		callCount++
    69  		env.urls[r.URL.Path]++
    70  
    71  		time.Sleep(delay)
    72  
    73  		if env.fetchFails {
    74  			w.WriteHeader(http.StatusBadRequest)
    75  			w.Write([]byte("fetch failed"))
    76  		} else {
    77  			features := FeatureMap{}
    78  			if encryptedFeatures == "" {
    79  				features = FeatureMap{"foo": {DefaultValue: *env.featureValue}}
    80  			}
    81  			response := &FeatureAPIResponse{
    82  				Features:          features,
    83  				EncryptedFeatures: encryptedFeatures,
    84  			}
    85  
    86  			if provideSSE {
    87  				w.Header().Set("X-SSE-Support", "enabled")
    88  			}
    89  			w.Header().Set("Content-Type", "application/json")
    90  			w.WriteHeader(http.StatusOK)
    91  			json.NewEncoder(w).Encode(response)
    92  		}
    93  	})
    94  
    95  	// SSE server handler.
    96  	if provideSSE {
    97  		env.sseServer = sse.New()
    98  		env.sseServer.CreateStream("features")
    99  		mux.HandleFunc("/sub/", env.sseServer.ServeHTTP)
   100  	}
   101  
   102  	env.server = httptest.NewServer(mux)
   103  
   104  	return &env
   105  }

   469  func TestRepoComplexSSEScenario(t *testing.T) {
   470  	env := setup(true)
   471  	defer cache.Clear()
   472  	// We're going to generate SSE errors here, but that's all we expect
   473  	// to see.
   474  	defer knownSSEErrors(t)
   475  	defer env.close()
   476  
   477  	cache.Clear()
   478  
   479  	// Data recording for test goroutines.
   480  	type record struct {
   481  		result string
   482  		t      time.Time
   483  	}
   484  
   485  	var wg sync.WaitGroup
   486  
   487  	// Test function to run in a goroutine: evaluates features at
   488  	// randomly spaced intervals, storing the results and the sample
   489  	// times, until told to stop.
   490  	tester := func(gb *GrowthBook, doneCh chan struct{}, vals *[]*record) {
   491  		defer wg.Done()
   492  		tick := time.NewTicker(time.Duration(100+rand.Intn(100)) * time.Millisecond)
   493  		gb.LoadFeatures(&FeatureRepoOptions{AutoRefresh: true})
   494  		for {
   495  			select {
   496  			case <-doneCh:
   497  				return
   498  
   499  			case <-tick.C:
   500  				f, _ := gb.EvalFeature("foo").Value.(string)
   501  				*vals = append(*vals, &record{f, time.Now()})
   502  			}
   503  		}
   504  	}
   505  
   506  	// Set up test goroutines, each with an independent GrowthBook
   507  	// instance, cancellation channel and result storage.
   508  	gbs := make([]*GrowthBook, 10)
   509  	doneChs := make([]chan struct{}, 10)
   510  	vals := make([][]*record, 10)
   511  	wg.Add(10)
   512  	for i := 0; i < 10; i++ {
   513  		gbs[i] = makeGB(env.server.URL, "qwerty1234", 0)
   514  		doneChs[i] = make(chan struct{})
   515  		vals[i] = []*record{}
   516  		go tester(gbs[i], doneChs[i], &vals[i])
   517  	}
   518  
   519  	// Command storage.
   520  	type command struct {
   521  		cmd int
   522  		t   time.Time
   523  	}
   524  	commands := make([]command, 100)
   525  
   526  	// Command loop: send SSE events at random intervals, with
   527  	// approximately 10% failure rate (and always at least three
   528  	// failures in a row, to trigger SSE client reconnection).
   529  	bad := 0
   530  	for i := 0; i < 100; i++ {
   531  		ok := rand.Intn(100) < 90
   532  		if ok && bad == 0 {
   533  			featuresJson := fmt.Sprintf(`{"features": {"foo": {"defaultValue": "val%d"}}}`, i+1)
   534  			commands[i] = command{i + 1, time.Now()}
   535  			env.sseServer.Publish("features", &sse.Event{Data: []byte(featuresJson)})
   536  		} else {
   537  			if bad == 0 {
   538  				bad = 3
   539  			} else {
   540  				bad--
   541  			}
   542  			commands[i] = command{-(i + 1), time.Now()}
   543  			env.sseServer.Publish("features", &sse.Event{Data: []byte("broken(bad")})
   544  		}
   545  		time.Sleep(time.Duration(50+rand.Intn(50)) * time.Millisecond)
   546  	}
   547  
   548  	// Stop the test goroutines and zero their GrowthBook instances so
   549  	// that finalizers will run and background SSE refresh will stop
   550  	// too.
   551  	for i := 0; i < 10; i++ {
   552  		doneChs[i] <- struct{}{}
   553  		gbs[i] = nil
   554  	}
   555  	wg.Wait()
   556  
   557  	// Check the results from the test goroutines by finding the
   558  	// relevant times in the command history. Allow some slack for small
   559  	// time differences.
   560  	errors := 0
   561  	for i := 0; i < 10; i++ {
   562  		for _, v := range vals[i] {
   563  			if v.result == "initial" {
   564  				continue
   565  			}
   566  			cmdidx, _ := sortFind(len(commands), func(i int) int {
   567  				if v.t == commands[i].t {
   568  					return 0
   569  				}
   570  				if v.t.After(commands[i].t) {
   571  					return 1
   572  				}
   573  				return -1
   574  			})
   575  
   576  			cmdidx--
   577  			expected := fmt.Sprintf("val%d", commands[cmdidx].cmd)
   578  
   579  			beforeidx := cmdidx - 1
   580  			for beforeidx > 0 && commands[beforeidx].cmd < 0 {
   581  				beforeidx--
   582  			}
   583  			before := fmt.Sprintf("val%d", commands[beforeidx].cmd)
   584  
   585  			afteridx := cmdidx + 1
   586  			for afteridx < len(commands)-1 && commands[afteridx].cmd < 0 {
   587  				afteridx++
   588  			}
   589  			after := ""
   590  			if afteridx < len(commands) {
   591  				after = fmt.Sprintf("val%d", commands[afteridx].cmd)
   592  			}
   593  
   594  			if v.result != expected && v.result != before && v.result != after {
   595  				errors++
   596  				t.Error("unexpected feature value")
   597  				fmt.Println(v.result, v.t, cmdidx, cmdidx, beforeidx, afteridx)
   598  			}
