    49  func setupWithDelay(provideSSE bool, delay time.Duration, encryptedFeatures string) *env {
    50  	SetLogger(&testLog)
    51  	testLog.reset()
    52  
    53  	initialValue := "initial"
    54  	callCount := 0
    55  	urls := map[string]int{}
    56  	env := env{featureValue: &initialValue, callCount: &callCount, urls: urls}
    57  
    58  	// We need to set up a mock server to handle normal API requests and
    59  	// SSE updates.
    60  	mux := http.NewServeMux()
    61  
    62  	// Normal GET.
    63  	mux.HandleFunc("/api/features/", func(w http.ResponseWriter, r *http.Request) {
    64  		env.Lock()
    65  		defer env.Unlock()
    66  		callCount++
    67  		env.urls[r.URL.Path]++
    68  
    69  		time.Sleep(delay)
    70  
    71  		if env.fetchFails {
    72  			w.WriteHeader(http.StatusBadRequest)
    73  			w.Write([]byte("fetch failed"))
    74  		} else {
    75  			features := FeatureMap{}
    76  			if encryptedFeatures == "" {
    77  				features = FeatureMap{"foo": {DefaultValue: *env.featureValue}}
    78  			}
    79  			response := &FeatureAPIResponse{
    80  				Features:          features,
    81  				DateUpdated:       time.Now(),
    82  				EncryptedFeatures: encryptedFeatures,
    83  			}
    84  
    85  			if provideSSE {
    86  				w.Header().Set("X-SSE-Support", "enabled")
    87  			}
    88  			w.Header().Set("Content-Type", "application/json")
    89  			w.WriteHeader(http.StatusOK)
    90  			json.NewEncoder(w).Encode(response)
    91  		}
    92  	})
    93  
    94  	// SSE server handler.
    95  	if provideSSE {
    96  		env.sseServer = sse.New()
    97  		env.sseServer.CreateStream("features")
    98  		mux.HandleFunc("/sub/", env.sseServer.ServeHTTP)
    99  	}
   100  
   101  	env.server = httptest.NewServer(mux)
   102  
   103  	return &env
   104  }

   219  func TestRepoUsesCacheAndCanRefreshManually(t *testing.T) {
   220  	env := setup(false)
   221  	defer cache.Clear()
   222  	defer checkLogs(t)
   223  	defer env.close()
   224  
   225  	cache.Clear()
   226  
   227  	// Set cache TTL short so we can test expiry.
   228  	gb := makeGB(env.server.URL, "qwerty1234", 100*time.Millisecond)
   229  	time.Sleep(20 * time.Millisecond)
   230  
   231  	// Initial value of feature should be null.
   232  	checkFeature(t, gb, "foo", nil)
   233  	env.checkCalls(t, 1)
   234  	knownWarnings(t, 1)
   235  
   236  	// Once features are loaded, value should be from the fetch request.
   237  	gb.LoadFeatures(nil)
   238  	checkFeature(t, gb, "foo", "initial")
   239  	env.checkCalls(t, 1)
   240  
   241  	// Value changes in API
   242  	*env.featureValue = "changed"
   243  
   244  	// New instances should get cached value
   245  	gb2 := makeGB(env.server.URL, "qwerty1234", 100*time.Millisecond)
   246  	checkFeature(t, gb2, "foo", nil)
   247  	knownWarnings(t, 1)
   248  	gb2.LoadFeatures(&FeatureRepoOptions{AutoRefresh: true})
   249  	checkFeature(t, gb2, "foo", "initial")
   250  
   251  	// Instance without autoRefresh.
   252  	gb3 := makeGB(env.server.URL, "qwerty1234", 100*time.Millisecond)
   253  	checkFeature(t, gb3, "foo", nil)
   254  	knownWarnings(t, 1)
   255  	gb3.LoadFeatures(nil)
   256  	checkFeature(t, gb3, "foo", "initial")
   257  
   258  	env.checkCalls(t, 1)
   259  
   260  	// Old instances should also get cached value.
   261  	checkFeature(t, gb, "foo", "initial")
   262  
   263  	// Refreshing while cache is fresh should not cause a new network
   264  	// request.
   265  	gb.RefreshFeatures(nil)
   266  	env.checkCalls(t, 1)
   267  
   268  	// Wait a bit for cache to become stale and refresh again.
   269  	time.Sleep(100 * time.Millisecond)
   270  	gb.RefreshFeatures(nil)
   271  	env.checkCalls(t, 2)
   272  
   273  	// The instance being updated should get the new value.
   274  	checkFeature(t, gb, "foo", "changed")
   275  
   276  	// The instance with auto-refresh should now have the new value.
   277  	checkFeature(t, gb2, "foo", "changed")
   278  
   279  	// The instance without auto-refresh should continue to have the old
   280  	// value.
   281  	checkFeature(t, gb3, "foo", "initial")
   282  
   283  	// New instances should get the new value
   284  	gb4 := makeGB(env.server.URL, "qwerty1234", 100*time.Millisecond)
   285  	checkFeature(t, gb4, "foo", nil)
   286  	knownWarnings(t, 1)
   287  	gb4.LoadFeatures(nil)
   288  	checkFeature(t, gb4, "foo", "changed")
   289  
   290  	env.checkCalls(t, 2)
   291  }
